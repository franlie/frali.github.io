# Fetch data from yfinance
# Timeframe: 10 Years daily data for Nasdaq ETF QQQ, S&P 500 ETF SPY, USD Index and 10 Year Treasuries

end = datetime.date(2023,8,2)
start = end - pd.Timedelta(days = 365*10)

# tickers= ['SPY', 'QQQ', 'DX-Y.NYB', '^TNX']
spy_df = yf.download("SPY", start=start, end=end, progress=False)
qqq_df = yf.download("QQQ", start=start, end=end, progress=False)
dollar_df = yf.download("DX-Y.NYB", start=start, end=end, progress=False)
treasuries_df = yf.download("^TNX", start=start, end=end, progress=False)

# Calculating Daily Returns (log)
spy_df_close = np.log(spy_df['Adj Close'] / spy_df['Adj Close'].shift(1))
qqq_df_close = np.log(qqq_df['Adj Close'] / qqq_df['Adj Close'].shift(1))
dollar_df_close = np.log(dollar_df['Adj Close'] / dollar_df['Adj Close'].shift(1))
treasuries_df_close = np.log(treasuries_df['Adj Close'] / treasuries_df['Adj Close'].shift(1))

# Create a dataframe for returns and remove NA
df = pd.concat([spy_df_close, qqq_df_close, dollar_df_close, treasuries_df_close], axis=1, keys=['SPY', 'QQQ', 'DOLLAR', 'TREASURY']).dropna()
